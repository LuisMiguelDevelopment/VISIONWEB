  <h2>Installation</h2>

  <ul>
    <li>SERVER</li>
    <ul>
      <li>npm install</li>
      <li>Create a file named <code>.env</code> with the following content:</li>
    </ul>
    <pre>
PORT=3001
DB_SERVER=root
DB_NAME=VISIONWEB
DB_USER=root
DB_PASSWORD=root
USER_EMAIL=
USER_PASS=
TOKEN_SECRET=
BASE_URL=http://localhost:3000/
    </pre>
    <ul>
      <li>npm run dev</li>
    </ul>
    <li>CLIENT</li>
    <ul>
      <li>npm install</li>
      <li>npm run dev</li>
    </ul>
